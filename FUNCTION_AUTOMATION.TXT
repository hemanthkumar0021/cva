CREATE OR REPLACE PROCEDURE UTIL_DB.UTIL_SCHEMA.GEN_SECURE_VIEW_DDL(
    SRC_DB STRING,
    SRC_SCHEMA STRING,
    SRC_TABLE STRING,
    TGT_DB STRING,
    TGT_SCHEMA STRING,
    TGT_VIEW STRING,
    COLUMN_LIST STRING
)
RETURNS STRING
LANGUAGE JAVASCRIPT
AS
$$
function escSqlString(str) {
    if (str === null || str === undefined) return '';
    return str.replace(/'/g, "''");
}

var maskedCols = {};
if (COLUMN_LIST !== null && COLUMN_LIST.trim() !== '') {
    COLUMN_LIST.split(',').forEach(function(col) {
        maskedCols[col.trim().toUpperCase()] = true;
    });
}

var metaSql = `
    SELECT
        COLUMN_NAME,
        COMMENT,
        ORDINAL_POSITION
    FROM ${SRC_DB}.INFORMATION_SCHEMA.COLUMNS
    WHERE TABLE_SCHEMA = '${SRC_SCHEMA.toUpperCase()}'
      AND TABLE_NAME   = '${SRC_TABLE.toUpperCase()}'
    ORDER BY ORDINAL_POSITION
`;

var stmt = snowflake.createStatement({sqlText: metaSql});
var rs = stmt.execute();

var colDdl = [];
var colSelect = [];

while (rs.next()) {
    var colName = rs.getColumnValue(1);
    var comment = rs.getColumnValue(2);
    var ordinal = rs.getColumnValue(3);

    var line = colName;
    if (maskedCols[colName.toUpperCase()]) {
        line += ' WITH MASKING POLICY POLICIES.CLS_PHI_STRING';
    }
    line += ` COMMENT '${escSqlString(comment)}'`;

    colDdl.push(line);
    colSelect.push(colName);
}

var ddl = `CREATE OR REPLACE SECURE VIEW ${TGT_DB}.${TGT_SCHEMA}.${TGT_VIEW}
(
${colDdl.join(',\n')}
)
COPY GRANTS
AS
SELECT
${colSelect.join(',\n')}
FROM ${SRC_DB}.${SRC_SCHEMA}.${SRC_TABLE};`;

return ddl;
$$;
